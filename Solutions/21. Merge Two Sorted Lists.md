## 解法1: 迭代
```
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
        if not l1:
            return l2
        if not l2: 
            return l1
        left, right = l1, l2
        res = ListNode()
        curr = res
        while left  and right :
            if left.val <= right.val:
                curr.next = left
                left = left.next
                curr = curr.next
            else:
                curr.next = right
                right = right.next
                curr = curr.next
                
        if left:
            curr.next = left
        if right:
            curr.next = right
        
        return res.next
        
```
