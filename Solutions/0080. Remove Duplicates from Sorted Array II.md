class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        j = 1
        count = 1
        for i in range(len(nums)):
            j = max(j, i + 1)
            
            while j < len(nums) and count > 1 and nums[i] == nums[j]:
                j += 1
            if j == len(nums):
                break
            if nums[i] == nums[j]:
                count += 1
            else:
                count = 1
            nums[i+1] = nums[j]
            j += 1
        return i + 1
