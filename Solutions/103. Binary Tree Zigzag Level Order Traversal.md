用BFS就可以实现，每层标记一下即可。
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def zigzagLevelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root:
            return []
       
        flag = False
        q = collections.deque()
        q.append(root)
        
        res = []
        
        while q:
            n = len(q)
            curr_level = []
            for _ in range(n):
                node = q.popleft()
                curr_level.append(node.val)
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
                    
            if flag:
                curr_level.reverse()
            res.append(curr_level)
            flag = not flag
            
        return res
```
