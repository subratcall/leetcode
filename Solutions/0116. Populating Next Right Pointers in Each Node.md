class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return
        self.connect(root.left)
        self.connect(root.right)
        p, q = root.left, root.right
        while p:
            p.next = q
            p = p.right
            q = q.left
        return root
