"""
# Definition for a Node.
class Node:
    def __init__(self, val, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right
"""

class Solution:
    def treeToDoublyList(self, root: 'Node') -> 'Node':
        head, tail = self.helper(root)
        return head
        
    def helper(self, root):
        if not root:
            return None, None

        left_head, left_tail = self.helper(root.left)
        right_head, right_tail = self.helper(root.right)
        
        if left_tail:
            left_tail.right = root
            root.left = left_tail
        else:
            left_head, left_tail = root, root
            
        if right_head:
            right_head.left = root
            root.right = right_head
        else:
            right_head, right_tail = root, root
            
        right_tail.right = left_head
        left_head.left = right_tail
        
        return left_head, right_tail

def dfs(self, root):
        if not root:
            return None, None
        left_head, left_tail = self.dfs(root.left)
        right_head, right_tail = self.dfs(root.right)
        if left_tail:
            left_tail.right = root
            root.left = left_tail
        if right_head:
            root.right = right_head
            right_head.left = root
            
        head = left_head or root
        tail = right_tail or root
        
        return head, tail
    
    def treeToDoublyList(self, root):
        if not root:
            return None
        head, tail = self.dfs(root)
        head.left = tail
        tail.right = head
        return head
