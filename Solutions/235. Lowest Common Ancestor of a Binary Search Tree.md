# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
        path_a = self.findPath(p.val)
        path_b = self.findPath(q.val)
        
        
        
    def findPath(self, root, target):
        if not root:
            return
        if root.val == target:
            return [root.val]
        if self.findPath(root.left, target):
            return root.val + self.findPath(root.left, target)
        else:
            return root.val + self.findPath(root.right, target)
