class Solution:
    def calculate(self, s: str) -> int:

        stack = []
        number = 0
        sign = '+'
        i = 0
        while i < len(s):
            if s[i] == ' ':
                i += 1
                continue
            while i < len(s) and s[i].isdigit():
                number = number * 10 + int(s[i])
                i += 1
            if sign == "+":
                stack.append(number)
            elif sign == "-":
                stack.append(-number)
            elif sign == "*":
                stack.append(stack.pop() * number)
            elif sign == "/":
                stack.append(int(stack.pop() / number))
            if i < len(s):
                sign = s[i]
            number = 0
            i += 1
        res = sum(stack)
        return res
