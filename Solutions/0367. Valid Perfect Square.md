class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        if num in (0, 1):
            return True
        left, right = 0, num // 2
        while left <= right:
            mid = (left + right) // 2
            
            if mid * mid == num:
                return True
            elif mid * mid < num:
                left = mid + 1
            else:
                right = mid - 1
        return False
