# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: ListNode, k: int) -> ListNode:
        if not head:
            return None
        length = 0
        p = head
        while p.next:
            length += 1
            p = p.next
        # p stops at the last node
        length += 1
            
        k = k % length
        
        if k == 0:
            return head
        
        tail = head
        for _ in range(length - k - 1):
            tail = tail.next
            
        new_head = tail.next
        tail.next = None
        p.next = head
        
        return new_head
