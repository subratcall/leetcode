class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:

        stack = []
        for i, ch in enumerate(s):
            if ch.isalpha():
                continue
            if ch == "(":
                stack.append([i, ch])

            elif not stack or stack[-1][1] != "(":
                stack.append([i, ch])
            else:
                stack.pop()
        
        to_del = set()
        for item in stack:
            to_del.add(item[0])
            
        res = ''
        for i in range(len(s)):
            if i not in to_del:
                res += s[i]
        
        return res
