# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        def convert2num(l):
            i = 0
            res = 0
            while l:
                res += l.val * 10 ** i
                i += 1
                l = l.next
            return res
        num1 = convert2num(l1)
        num2 = convert2num(l2)
        num3 = num1 + num2
        
        digits = list(str(num3))
        
        dummy = ListNode(0)
        p = dummy
        
        while digits:
            digit = int(digits.pop())
            new_node = ListNode(digit)
            p.next = new_node
            p = p.next
        p.next = None
        return dummy.next
