class Solution:
    def decodeString(self, s: str) -> str:
        stack = []
        number = 0
        
        for ch in s:
            if ch.isalpha():
                stack.append(ch)
            elif ch.isnumeric():
                number = number * 10 + int(ch)
            elif ch == "[":
                stack.append(number)
                number = 0
            else:
                buffer = []
                while isinstance(stack[-1], str):
                    buffer.append(stack.pop())
                num = stack.pop()
                
                res = buffer * num
                res.reverse()
                stack.append(''.join(res))
                
        res = ''.join(stack)
        return res
