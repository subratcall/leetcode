class WordDictionary:

    def __init__(self):
        self.end = False
        self.children = {}
        
    def addWord(self, word: str) -> None:
        p = self
        for ch in word:
            if ch not in p.children:
                p.children[ch] = WordDictionary()
            p = p.children[ch]
        p.end = True
        
    def search(self, word: str) -> bool:
        if word == "":
            return True
        p = self
        for i in range(len(word)):
            ch = word[i]
            if ch == ".":
                for child in p.children:
                    if p.children[child].search(word[i+1:]):
                        return True
                return False
            else:
                if ch not in p.children:
                    return False
                p = p.children[ch]
        return p.end
